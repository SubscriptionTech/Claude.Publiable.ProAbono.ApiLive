## sidebars.js

```js
// website/sidebars.js
import apiSidebar from './docs/api-reference/sidebar.ts';

/** @type {import('@docusaurus/plugin-content-docs').SidebarsConfig} */

  docsSidebar: [
    {type: 'autogenerated', dirName: 'introduction'},
    {type: 'autogenerated', dirName: 'quick-start'},
    {type: 'autogenerated', dirName: 'authentication'},
    {type: 'autogenerated', dirName: 'concepts'},
    ...apiSidebar,
    {type: 'autogenerated', dirName: 'guides'},
  ],
};

export default sidebars;
```
